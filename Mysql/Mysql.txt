-- Create users table
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    phone VARCHAR(20),

-- Create listings table
CREATE TABLE IF NOT EXISTS listings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    title VARCHAR(200) NOT NULL,
    description TEXT NOT NULL,
    location VARCHAR(150) NOT NULL,
    price_per_night DECIMAL(8,2) NOT NULL,
    max_guests INT NOT NULL,
    bedrooms INT DEFAULT 1,
    beds INT NOT NULL,
    bathrooms DECIMAL(3,1) DEFAULT 1,
    image_url VARCHAR(500) NOT NULL,
    rating DECIMAL(3,2) DEFAULT 4.5,
    review_count INT DEFAULT 0,
    is_guest_favorite BOOLEAN DEFAULT FALSE,
    instant_book BOOLEAN DEFAULT TRUE,
    free_cancellation BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Create bookings table

    FOREIGN KEY (listing_id) REFERENCES listings(id) ON DELETE CASCADE
);

-- Create reviews table
CREATE TABLE IF NOT EXISTS reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    listing_id INT NOT NULL,
    rating DECIMAL(2,1) NOT NULL,
    comment TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (listing_id) REFERENCES listings(id) ON DELETE CASCADE
);

-- Get database statistics
SELECT 
    (SELECT COUNT(*) FROM users) as total_users,
    (SELECT COUNT(*) FROM listings) as total_properties,
    (SELECT COUNT(*) FROM bookings) as total_bookings,
    (SELECT COUNT(*) FROM chat_logs) as total_chats;

-- Get property statistics by location
SELECT location, COUNT(*) as count, AVG(price_per_night) as avg_price
FROM listings
GROUP BY location
ORDER BY count DESC;

-- Get monthly booking statistics
SELECT 
    DATE_FORMAT(created_at, '%Y-%m') as month,
    COUNT(*) as bookings_count,
    SUM(total_price) as total_revenue
FROM bookings
GROUP BY DATE_FORMAT(created_at, '%Y-%m')
ORDER BY month DESC;
